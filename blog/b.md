```js
import axios from 'axios';
//用于请求数据，get 方法 post request response

function getJson(){
  let address=`https://raw.githubusercontent.com/xixilide/demoData/master/demo.json
  ?${Math.random()}`;
  //?${Math.random()用于查询
  return axios.get(address)
    .then((res) => ({

      getJson:res.data
    }))
    .catch(function (error) {
      alert(error);
    })
}
export { getJson };

```
